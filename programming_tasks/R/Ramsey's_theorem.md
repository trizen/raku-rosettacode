# [Ramsey's theorem][1]

```perl6
my @a = [ 0 xx 17 ] xx 17;
@a[$_;$_] = '-' for ^17;
 
for ^17 X 1,2,4,8 -> $i, $k {
    my $j = ($i + $k) % 17;
    @a[$i;$j] = @a[$j;$i] = 1;
}
.say for @a;
 
for combinations(17,4) -> $quartet {
    my $links = [+] $quartet.combinations(2).map: -> [$i;$j] { @a[$i;$j] }
    die "Bogus!" unless 0 < $links < 6;
}
say "OK";
```
